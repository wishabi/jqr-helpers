def copy_files
  log 'Copying files...'
  files = [
    'javascripts/jqr-helpers.js',
    'stylesheets/jqr-helpers.css',
    'images/jqr-helpers/close.png',
    'images/jqr-helpers/throbber.gif'
  ]
  files.each do |file|
    copy_file file, "public/#{file}"
  end
end
