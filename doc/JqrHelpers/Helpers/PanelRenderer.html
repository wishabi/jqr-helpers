def panel(title, url_or_options={}, options={}, &block)
  if url_or_options.is_a?(String)
    url = url_or_options
    content = ''
    id = nil
  else
    options = url_or_options
    content = yield
    id = Helpers._random_string
    url = '#' + id
  end
  options.merge!(:id => id)
  panels << {
    :title => title,
    :url => url,
    :options => options,
    :content => content
  }
  nil # suppress output
end
